\begin{frame}{Getting the Data from Coincall}
\begin{itemize}
  \item Base URL: https://api.coincall.com   |   Python SDK: github.com/CoincallExchange
  \item Signed request headers: X-CC-APIKEY, sign (HMAC-SHA256, upper hex), ts (ms), X-REQ-TS-DIFF
  \item Signing: sort params alphabetically, append uuid, ts, x-req-ts-diff, HMAC-SHA256 with secret
  \item Rate limits (per docs): orderbook 30/2s per user; public config 10/2s per IP
  \item Key endpoints you will use across the program:
  \begin{itemize}\item \texttt{\small Instruments:   GET /open/option/getInstruments/\{baseCurrency\}}\end{itemize}
  \begin{itemize}\item \texttt{\small Option chain:  GET /open/option/get/v1/\{index\}}\end{itemize}
  \begin{itemize}\item \texttt{\small Option book:   GET /open/option/order/orderbook/v1/\{symbol\}  (100 depth)}\end{itemize}
  \begin{itemize}\item \texttt{\small Option detail: GET /open/option/detail/v1/\{symbol\}  (mark, IV, delta/gamma/vega/theta)}\end{itemize}
  \begin{itemize}\item \texttt{\small Trades:        GET /open/option/trade/lasttrade/v1/\{symbol\}}\end{itemize}
  \begin{itemize}\item \texttt{\small Futures book:  GET /open/futures/order/orderbook/v1/\{symbol\}}\end{itemize}
  \begin{itemize}\item \texttt{\small Funding:       GET /open/public/fundingRate/v1?symbol=BTCUSD}\end{itemize}
  \begin{itemize}\item \texttt{\small Klines:        GET /open/option/market/kline/history/v1/\{optionName\}?period=h1}\end{itemize}
  \item WebSocket channels: /options/pricing, /options/orderbook, /options/lasttrade, /options/kline
  \item ALWAYS verify the exact path/params against https://docs.coincall.com/ -- the API evolves.
\end{itemize}
\vspace{2pt}\tiny\textit{Cross-check exact paths at https://docs.coincall.com/}
\end{frame}

\begin{frame}[fragile]{Code --- coincall\_client.py}
\begin{lstlisting}
import time, hmac, hashlib, uuid, requests

BASE = "https://api.coincall.com"

def signed_get(path, params=None, api_key="", api_secret=""):
    """Minimal Coincall signed GET. Public market-data endpoints often work
    unsigned; this shows the signed pattern used for account endpoints."""
    params = params or {}
    ts, window = str(int(time.time() * 1000)), "5000"
    items = sorted(params.items())                       # sort alphabetically
    payload = "&".join(f"{k}={v}" for k, v in items)
    payload += f"&uuid={uuid.uuid4().hex}&ts={ts}&x-req-ts-diff={window}"
    sign = hmac.new(api_secret.encode(), payload.encode(),
                    hashlib.sha256).hexdigest().upper()
    headers = {"X-CC-APIKEY": api_key, "sign": sign,
               "ts": ts, "X-REQ-TS-DIFF": window}
    r = requests.get(BASE + path, params=params, headers=headers, timeout=10)
    r.raise_for_status()
    return r.json()

# Example: list all BTC option instruments
print(signed_get("/open/option/getInstruments/BTC"))
\end{lstlisting}
\end{frame}
\begin{frame}{Glossary --- Key Terms}
\footnotesize\begin{description}
\item[\color{ncsured}Market maker] liquidity provider that continuously quotes bid and ask, earning the spread
\item[\color{ncsured}Taker] participant that crosses the spread for immediate execution
\item[\color{ncsured}Bid-ask spread] best ask minus best bid; the maker's revenue and risk premium
\item[\color{ncsured}Inventory] the maker's net position; the main source of risk
\item[\color{ncsured}Milestone] graded software deliverable, due in Weeks 5, 7, and 9
\item[\color{ncsured}PyTorch / autograd] Python ML library; automatic differentiation computes Greeks for you
\item[\color{ncsured}pybind11] library that exposes C++ functions to Python (used in Week 7)
\end{description}
