\section*{Derivation}
\begin{frame}{Derivation \& Rationale}
\[dS_t = \sigma\, dW_t\]

\small\textbf{\color{ncsured}Why:} Arithmetic (not geometric) Brownian motion is chosen because over a short market-making horizon price changes are small and roughly additive; it keeps the HJB linear in $S$ and makes the cash dynamics exact.
\end{frame}

\begin{frame}{Derivation \& Rationale}
\[\lambda(\delta) = A\, e^{-\kappa \delta}\]

\small\textbf{\color{ncsured}Why:} Fill intensity decays with quote distance $\delta$. The exponential form is the simplest law consistent with 'further from mid $\Rightarrow$ fewer fills'; $A$ sets the base rate, $\kappa$ the decay. It also yields a closed-form optimum.
\end{frame}

\begin{frame}{Derivation \& Rationale}
\[\max\ \mathbb{E}\!\left[-e^{-\gamma\,(X_T + q_T S_T)}\right]\]

\small\textbf{\color{ncsured}Why:} CARA (exponential) utility is used because constant absolute risk aversion makes the value function factor out wealth, so the optimal quotes do not depend on the cash level -- exactly what we want for a market maker.
\end{frame}

\begin{frame}{Derivation \& Rationale}
\[\partial_t V + \tfrac12\sigma^2 \partial_{SS} V + \max_{\delta^b}\lambda(\delta^b)\big[V(q{+}1,x{-}S{+}\delta^b)-V\big]+ \max_{\delta^a}\lambda(\delta^a)\big[V(q{-}1,x{+}S{+}\delta^a)-V\big]=0\]

\small\textbf{\color{ncsured}Why:} The HJB equation: the diffusion term prices inventory risk; each $\max$ term is the dealer optimally choosing the bid/ask offset, trading fill probability against captured spread.
\end{frame}

\begin{frame}{Derivation \& Rationale}
\[V(t,x,S,q) = -e^{-\gamma(x+qS)}\,\theta(t,q)\]

\small\textbf{\color{ncsured}Why:} This ansatz is not a guess from nowhere: under CARA $+$ arithmetic dynamics the value function inherits a multiplicative structure across $(x, qS)$, so all the wealth/price dependence is exponential and only $\theta(t,q)$ remains to solve.
\end{frame}

\begin{frame}{Derivation \& Rationale}
\[r(t,q) = S - q\,\gamma\,\sigma^2 (T-t)\]

\small\textbf{\color{ncsured}Why:} The reservation (indifference) price: it sits BELOW mid when you are long ($q>0$) so you quote to sell down inventory. The skew scales with risk aversion $\gamma$, variance $\sigma^2$, and time left -- more risk $\Rightarrow$ more skew.
\end{frame}

\begin{frame}{Derivation \& Rationale}
\[\delta^* = \tfrac{1}{\gamma}\ln\!\big(1+\tfrac{\gamma}{\kappa}\big) + \tfrac{\gamma\sigma^2}{2}(T-t)\]

\small\textbf{\color{ncsured}Why:} The optimal half-spread = an order-flow term $\tfrac1\gamma\ln(1+\gamma/\kappa)$ (how hard fills are to get) plus an inventory-risk term $\tfrac{\gamma\sigma^2}{2}(T-t)$. You widen when risk-averse, when volatile, and when far from the horizon.
\end{frame}

import torch

def as_quotes(S, q, gamma, sigma, kappa, tau):
    """Asymptotic Avellaneda-Stoikov bid/ask. Vectorized over inventory q."""
    reservation = S - q * gamma * sigma**2 * tau
    half_spread = (1.0/gamma) * torch.log1p(gamma/kappa) \
                  + 0.5 * gamma * sigma**2 * tau
    return reservation - half_spread, reservation + half_spread  # bid, ask

S = torch.tensor(100.0); q = torch.arange(-10, 11).float()
bid, ask = as_quotes(S, q, gamma=0.1, sigma=2.0, kappa=1.5, tau=1.0)
print("inventory ->", q.tolist())
print("bid       ->", bid.round(decimals=2).tolist())
\end{lstlisting}

\begin{frame}{Glossary --- Key Terms}
\footnotesize\begin{description}
\item[\color{ncsured}Reservation price] inventory-adjusted indifference price that quotes are centered on
\item[\color{ncsured}Half-spread] distance from the reservation price to each quote
\item[\color{ncsured}HJB equation] the PDE characterizing the optimal value function
\item[\color{ncsured}Ansatz] an assumed functional form that reduces the HJB to ODEs
\item[\color{ncsured}CARA utility] constant absolute risk aversion, U(w) = -exp(-gamma w)
\item[\color{ncsured}Intensity lambda(delta)] Poisson fill rate as a function of quote distance delta
\item[\color{ncsured}Risk aversion gamma] parameter setting inventory skew and spread width
\end{description}
\end{frame}
