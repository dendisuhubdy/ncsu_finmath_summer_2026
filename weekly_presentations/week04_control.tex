\section*{Derivation}
\begin{frame}{Derivation \& Rationale}
\[V(t,x)=\sup_{u}\ \mathbb{E}\!\left[\int_t^T f(s,X_s,u_s)\,ds + g(X_T)\right]\]

\small\textbf{\color{ncsured}Why:} The value function is the best achievable expected reward from $(t,x)$ onward. We define the whole problem through $V$ because Bellman's principle then gives a local (differential) equation instead of an intractable global optimization.
\end{frame}

\begin{frame}{Derivation \& Rationale}
\[V(t,x)=\sup_u \mathbb{E}\big[V(t{+}dt, X_{t+dt}) + f\,dt\big]\]

\small\textbf{\color{ncsured}Why:} The dynamic programming principle: an optimal policy is optimal from every later state too. Expanding $V(t+dt,\cdot)$ with Ito's lemma turns this into the HJB PDE.
\end{frame}

\begin{frame}{Derivation \& Rationale}
\[\partial_t V + \sup_{u}\big\{\mathcal{L}^u V + f(t,x,u)\big\}=0,\quad V(T,x)=g(x)\]

\small\textbf{\color{ncsured}Why:} The HJB equation. $\mathcal{L}^u$ is the generator of the controlled diffusion; the $\sup$ is the instantaneous optimal control. This is the engine behind every model in the course -- Avellaneda-Stoikov is one special case.
\end{frame}

\begin{frame}{Derivation \& Rationale}
\[\text{CARA: } U(w)=-e^{-\gamma w}\ \Rightarrow\ \text{policy independent of } w\]

\small\textbf{\color{ncsured}Why:} Constant absolute risk aversion makes marginal utility scale-free in wealth, so the optimal quotes do not move with the cash balance. CRRA ($U=w^{1-\gamma}/(1-\gamma)$) instead makes the policy scale WITH wealth -- natural for investment, awkward for a desk.
\end{frame}

\begin{lstlisting}
import torch
# Merton under CARA, solved by backward value iteration on a wealth grid.
w = torch.linspace(-5, 5, 401); dt = 1e-3; mu, r, sig, gam = 0.08, 0.02, 0.2, 1.0
V = -torch.exp(-gam * w)                      # terminal utility
for _ in range(1000):                         # march backward in time
    # optimal risky holding pi* = (mu - r)/(gam*sig^2) is wealth-independent (CARA)
    pi = (mu - r) / (gam * sig**2)
    drift = (r*w + pi*(mu - r))
    diff  = 0.5 * (pi*sig)**2
    dV = torch.gradient(V, spacing=(w[1]-w[0]).item())[0]
    d2V = torch.gradient(dV, spacing=(w[1]-w[0]).item())[0]
    V = V + dt * (drift*dV + diff*d2V)
print("pi* (constant in wealth):", (mu - r)/(gam*sig**2))
\end{lstlisting}
\end{frame}
\begin{frame}{Glossary --- Key Terms}
\footnotesize\begin{description}
\item[\color{ncsured}Stochastic optimal control] choosing controls to optimize an expected objective under randomness
\item[\color{ncsured}Dynamic programming principle] an optimal policy stays optimal from every later state
\item[\color{ncsured}Value function] the best achievable expected reward from a given state onward
\item[\color{ncsured}Infinitesimal generator] operator giving the expected instantaneous change of a function
\item[\color{ncsured}Verification theorem] conditions under which an HJB solution is the value function
\item[\color{ncsured}CRRA utility] constant relative risk aversion; the policy scales with wealth
\item[\color{ncsured}Admissible control] a control satisfying the problem's constraints
\end{description}
\end{frame}
