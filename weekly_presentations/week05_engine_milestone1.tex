\section*{Derivation}
\begin{frame}{Derivation \& Rationale}
\[\mathcal{L}(A,\kappa)=\prod_i \lambda(\delta_i)\,e^{-\lambda(\delta_i)\,\Delta t_i}\]

\small\textbf{\color{ncsured}Why:} Fills are a (doubly stochastic) Poisson process, so the likelihood of the observed fills given quote distances $\delta_i$ is a product of Poisson densities. We fit by MAXIMUM LIKELIHOOD rather than least-squares because the data are event counts, not Gaussian residuals.
\end{frame}

\begin{frame}{Derivation \& Rationale}
\[-\ln\mathcal{L} = \sum_i\big[\lambda(\delta_i)\Delta t_i - \ln\lambda(\delta_i)\big]\]

\small\textbf{\color{ncsured}Why:} Minimizing this negative log-likelihood is convex in $\ln A$ and well-behaved in $\kappa$; PyTorch autograd gives the gradients, so a few hundred Adam steps converge.
\end{frame}

\begin{frame}[fragile]{Code --- week5\_calibrate.py}
\begin{lstlisting}
import torch
# delta: quote distances at which fills occurred; dt: exposure time per fill
delta = torch.tensor([...]); dt = torch.tensor([...])
logA  = torch.zeros(1, requires_grad=True)
kappa = torch.ones(1,  requires_grad=True)
opt = torch.optim.Adam([logA, kappa], lr=0.05)
for step in range(2000):
    lam = torch.exp(logA) * torch.exp(-kappa.clamp(min=1e-3) * delta)
    nll = (lam * dt - torch.log(lam)).sum()          # Poisson negative log-lik
    opt.zero_grad(); nll.backward(); opt.step()
print("A =", torch.exp(logA).item(), " kappa =", kappa.item())
\end{lstlisting}
\end{frame}
\begin{frame}{Glossary --- Key Terms}
\footnotesize\begin{description}
\item[\color{ncsured}Maximum likelihood (MLE)] fitting parameters by maximizing the likelihood of the data
\item[\color{ncsured}Poisson process] counting process for random arrivals (here, fills)
\item[\color{ncsured}Goodness-of-fit (chi-square)] a statistic measuring how well the fit matches the data
\item[\color{ncsured}Backtest] simulation of a strategy on historical data
\item[\color{ncsured}Fill simulator] model of which of the maker's quotes get executed
\item[\color{ncsured}P\&L attribution] decomposition of profit into its sources
\item[\color{ncsured}Reproducibility] identical results on re-run under a fixed random seed
\end{description}
\end{frame}
