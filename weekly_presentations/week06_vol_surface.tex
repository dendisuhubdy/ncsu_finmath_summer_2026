\section*{Derivation}
\begin{frame}{Derivation \& Rationale}
\[w(k)=a+b\big(\rho(k-m)+\sqrt{(k-m)^2+\varsigma^2}\big)\]

\small\textbf{\color{ncsured}Why:} Raw SVI for total implied variance $w=\sigma^2 T$ in log-moneyness $k$. The hyperbola shape is chosen because it matches observed smiles with only 5 parameters and has closed-form slopes for the no-arbitrage checks.
\end{frame}

\begin{frame}{Derivation \& Rationale}
\[\text{butterfly: } g(k)\ge 0,\qquad \text{calendar: } \partial_T w(k,T)\ge 0\]

\small\textbf{\color{ncsured}Why:} These inequalities are exactly 'no free option-portfolio arbitrage': a negative $g(k)$ implies a negative risk-neutral density, and decreasing total variance in $T$ implies a calendar arbitrage. We enforce them so the fitted surface is tradeable.
\end{frame}

\begin{frame}[fragile]{Code --- week6\_bs\_autograd.py}
\begin{lstlisting}
import torch
def bs_call(S, K, T, r, sig):
    d1 = (torch.log(S/K) + (r + 0.5*sig**2)*T) / (sig*torch.sqrt(T))
    d2 = d1 - sig*torch.sqrt(T)
    N = torch.distributions.Normal(0., 1.).cdf
    return S*N(d1) - K*torch.exp(-r*T)*N(d2)

S,K,T,r,sig = [torch.tensor(x, requires_grad=True)
               for x in (100., 100., 1.0, 0.0, 0.6)]   # crypto IV ~ 60%
price = bs_call(S,K,T,r,sig)
vega,  = torch.autograd.grad(price, sig, create_graph=True)
vanna, = torch.autograd.grad(vega, S, create_graph=True)   # d vega / dS
volga, = torch.autograd.grad(vega, sig)                    # d vega / d sig
print(float(price), float(vega), float(vanna), float(volga))
\end{lstlisting}

\begin{frame}{Glossary --- Key Terms}
\footnotesize\begin{description}
\item[\color{ncsured}Implied volatility] the volatility that matches the market option price under Black-Scholes
\item[\color{ncsured}Volatility smile] implied volatility as a function of strike / moneyness
\item[\color{ncsured}SVI] Stochastic Volatility Inspired parameterization of total variance
\item[\color{ncsured}SABR] stochastic alpha-beta-rho model with the Hagan implied-vol expansion
\item[\color{ncsured}Greeks] price sensitivities: delta, gamma, vega, theta, rho
\item[\color{ncsured}Vanna / Volga] second-order Greeks: d vega / dS and d vega / d sigma
\item[\color{ncsured}Calendar / butterfly arbitrage] no-arbitrage constraints the surface must satisfy
\end{description}
\end{frame}
