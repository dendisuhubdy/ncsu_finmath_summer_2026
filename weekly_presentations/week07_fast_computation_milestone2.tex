\section*{Derivation}
\begin{frame}{Derivation \& Rationale}
\[C(K)=e^{-rT}\,\tfrac{1}{\pi}\!\int_0^\infty \mathrm{Re}\!\big[e^{-iu\ln K}\hat{\phi}(u)\big]\,du\]

\small\textbf{\color{ncsured}Why:} Once we leave Black-Scholes (e.g. Heston), there is no closed-form price, but the characteristic function $\phi$ IS known. Pricing becomes a single Fourier integral -- this is why CF methods exist and why they are fast.
\end{frame}

\begin{frame}{Derivation \& Rationale}
\[C \approx \sum_{n=0}^{N-1}{}' \mathrm{Re}\!\Big[\phi\!\big(\tfrac{n\pi}{b-a}\big)e^{-i n\pi\frac{a}{b-a}}\Big]V_n\]

\small\textbf{\color{ncsured}Why:} The Fang-Oosterlee COS method replaces the integral with a cosine series that converges geometrically for smooth densities. We truncate $[a,b]$ from the cumulants, not by guessing -- that is what makes it reach $10^{-4}$ accuracy in microseconds.
\end{frame}

\begin{frame}[fragile]{Code --- fastmm pybind11 module}
\begin{lstlisting}
// fastmm.cpp  (build with CMake + pybind11; import as `fastmm`)
#include <pybind11/pybind11.h>
#include <pybind11/numpy.h>
namespace py = pybind11;

double cos_price(double S, double K, double T, double r, double sigma);  // COS pricer

py::array_t<double> price_surface(py::array_t<double> K, double S,
                                  double T, double r, double sigma) {
    auto k = K.unchecked<1>(); auto out = py::array_t<double>(k.shape(0));
    auto o = out.mutable_unchecked<1>();
    { py::gil_scoped_release nogil;                 // release GIL for the hot loop
      for (py::ssize_t i = 0; i < k.shape(0); ++i)
          o(i) = cos_price(S, k(i), T, r, sigma); }
    return out;                                     // zero-copy back to NumPy
}
PYBIND11_MODULE(fastmm, m) { m.def("price_surface", &price_surface); }
\end{lstlisting}
\end{frame}
\begin{frame}{Glossary --- Key Terms}
\footnotesize\begin{description}
\item[\color{ncsured}Characteristic function] the Fourier transform of the log-return density
\item[\color{ncsured}COS method] Fourier-cosine series option pricing (Fang-Oosterlee)
\item[\color{ncsured}Carr-Madan] FFT-based option pricing via a damped payoff
\item[\color{ncsured}Implied-vol inversion] solving for the IV that reproduces a price (safeguarded Newton)
\item[\color{ncsured}pybind11] header-only C++ \textless{}-\textgreater{} Python binding
\item[\color{ncsured}GIL] Python's Global Interpreter Lock; released around the C++ hot loop
\item[\color{ncsured}Latency budget] the maximum allowed time (sub-5 ms full-surface revaluation)
\end{description}
\end{frame}
