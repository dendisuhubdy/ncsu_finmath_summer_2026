\section*{Derivation}
\begin{frame}{Derivation \& Rationale}
\[\widehat{SR}=\frac{\bar r}{s_r}\sqrt{252},\quad \text{CI via block bootstrap on } \{r_t\}\]

\small\textbf{\color{ncsured}Why:} Daily Sharpe is annualized by $\sqrt{252}$, but a point estimate over a few months is noisy and serially correlated. We resample BLOCKS of returns (preserving autocorrelation) to get an honest confidence interval -- a high Sharpe with a wide CI is not a real edge.
\end{frame}

import numpy as np
def attribute(spread, inventory, gamma, hedging):
    total = spread + inventory + gamma + hedging
    assert abs(total.sum() - (spread.sum()+inventory.sum()
               +gamma.sum()+hedging.sum())) < 1e-6           # must sum to total
    return dict(spread=spread, inventory=inventory, gamma=gamma,
                hedging=hedging, total=total)

def bootstrap_sharpe(r, block=5, n=2000, seed=0):
    rng = np.random.default_rng(seed); T = len(r); out = []
    for _ in range(n):
        idx = (rng.integers(0, T-block, T//block)[:,None] + np.arange(block)).ravel()
        s = r[idx]; out.append(s.mean()/s.std()*np.sqrt(252))
    return np.percentile(out, [2.5, 50, 97.5])               # CI for the Sharpe
\end{lstlisting}

\begin{frame}{Glossary --- Key Terms}
\footnotesize\begin{description}
\item[\color{ncsured}Event-driven backtest] a simulation that replays timestamped events in order
\item[\color{ncsured}Deterministic replay] bitwise-identical results on re-run
\item[\color{ncsured}Look-ahead bias] using information not available at decision time
\item[\color{ncsured}Sharpe ratio] risk-adjusted return: annualized mean over standard deviation
\item[\color{ncsured}Block bootstrap] resampling blocks of returns for a CI that keeps autocorrelation
\item[\color{ncsured}Maximum drawdown] the largest peak-to-trough decline in equity
\item[\color{ncsured}Taker strategy] aggressive, spread-crossing execution
\end{description}
