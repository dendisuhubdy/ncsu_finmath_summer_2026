\section*{Derivation}
\begin{frame}{Derivation \& Rationale}
\[\min_{\delta}\ \max_{\Sigma\in\mathcal{U}}\ \Big\{\text{spread P\&L} - \tfrac12 q^{\mathsf T}\Sigma q\Big\}\]

\small\textbf{\color{ncsured}Why:} Robust market making: we do not trust a single covariance estimate $\Sigma$, so we optimize against the WORST case over an uncertainty set $\mathcal{U}$. This widens quotes exactly where parameter risk is highest -- a principled cure for over-fitting the penalty matrix.
\end{frame}

import numpy as np
def collar_payoff(S, spot=100.0, put_K=90.0, call_K=110.0):
    """Long underlying + long put(put_K) + short call(call_K): a fixed band."""
    return (S - spot) + np.maximum(put_K - S, 0.0) - np.maximum(S - call_K, 0.0)

S = np.linspace(60, 140, 9)
print(dict(zip(S.round(0), collar_payoff(S).round(1))))   # capped above, floored below
\end{lstlisting}

\begin{frame}{Glossary --- Key Terms}
\footnotesize\begin{description}
\item[\color{ncsured}Multi-asset market making] jointly quoting correlated underlyings (e.g., BTC and ETH)
\item[\color{ncsured}Robust optimization] optimizing against the worst case over an uncertainty set
\item[\color{ncsured}Adverse-selection-aware quoting] skewing quotes using order-flow signals
\item[\color{ncsured}Collar] long put + short call that fixes a price band around a holding
\item[\color{ncsured}Accumulator / decumulator] products for staged accumulation or liquidation
\item[\color{ncsured}Coupon product] a yield-enhancement structured note
\item[\color{ncsured}Structured product] a packaged derivative payoff sold to clients or corporates
\end{description}
\end{frame}
